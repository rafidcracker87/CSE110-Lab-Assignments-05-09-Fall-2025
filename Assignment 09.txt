import java.util.Scanner;
public class lab_09_01{
    static int fact=1;
    static int factorial(int num){
        if(num==0 || num==1){
            return 1;
        }
        return num*factorial(num-1);
    }
     public static void main(String[] args){
     Scanner sc=new Scanner(System.in);
     System.out.print("Enter an integer:");
     int num1=sc.nextInt();
     System.out.println(factorial(num1));
     }
}

import java.util.Scanner;
public class lab_09_02{
    static int power(int base,int exp){
        if(exp==0){
            return 1;
        }
        else{
            return base*power(base,exp-1);
        }
    }
 public static void main(String[] args){
     Scanner sc=new Scanner(System.in);
     System.out.print("Enter an integer:");
     int num=sc.nextInt();
     System.out.print("Enter the exp:");
     int exp1=sc.nextInt();
     System.out.println(power(num,exp1));
     }
}

import java.util.Scanner;
public class lab_09_03{
    static void print_element(int[] arr,int index){
        if(index<arr.length){
        System.out.println(arr[index]);
        print_element(arr,index+1);
        }
    }
public static void main(String[] args){
     Scanner sc=new Scanner(System.in);
     System.out.print("Enter an index:");
     int index1=sc.nextInt();
     System.out.print("Enter the length of array:");
     int len=sc.nextInt();
     int[] arr1=new int[len];
     for(int i=0;i<len;i++){
         System.out.print("Enter a number:");
         arr1[i]=sc.nextInt();
     }
     print_element(arr1,index1);
}
}

import java.util.Scanner;
public class lab_09_04{
    static int fibonacci(int num){
        if(num==0){
            return 0;
        }
        if(num==1){
            return 1;
        }
        int prev_first=fibonacci(num-1);
        int prev_second=fibonacci(num-2);
        return prev_first+prev_second;
    }
    public static void main(String[] args){
     Scanner sc=new Scanner(System.in);
     System.out.print("Enter the index:");
     int num1=sc.nextInt();
     System.out.println(fibonacci(num1));
    }
}
